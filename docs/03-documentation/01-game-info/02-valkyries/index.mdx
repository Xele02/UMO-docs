export const ValkyrieClick = ({ids}) => {

    const list = ids.map(k => {
        const url = getLinkById("documentation_game_valkyrie_"+k.toString());
        return (
            <article class="col col--3 margin-bottom--lg">
            <Link href={url} className={clsx('card ', styles.cardContainer)}>
                <ValkyrieImage valkId={k} /><br />
                <TranslatedString bank="master" id={"vn_" + k.toString().padStart(4, '0')} language="ja" />
            </Link>
            </article>
    );
})

    return <>{list}</>;
}
